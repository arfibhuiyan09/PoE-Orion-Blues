About

A VEX V5 robot programmed to physically strike a snare drum and kick drum in 4/4 time at 120 BPM — replicating a kick-snare-kick-snare pattern from Twinkle Twinkle Little Star. In v.1.3.0, an optical sensor controls playback: light present = drums play, darkness = drums stop.

#region VEXcode Generated Robot Configuration
from vex import *
import urandom
import math

# Brain should be defined by default
brain=Brain()

# Robot configuration code
motor_1 = Motor(Ports.PORT1, GearSetting.RATIO_6_1, True)
motor_10 = Motor(Ports.PORT10, GearSetting.RATIO_6_1, False)
sensor = Optical(Ports.PORT2)

# wait for rotation sensor to fully initialize
wait(30, MSEC)

# Make random actually random
def initializeRandomSeed():
    wait(100, MSEC)
    random = brain.battery.voltage(MV) + brain.battery.current(CurrentUnits.AMP) * 100 + brain.timer.system_high_res()
    urandom.seed(int(random))

# Set random seed
initializeRandomSeed()

# add a small delay to make sure we don't print in the middle of the REPL header
wait(200, MSEC)
# clear the console to make sure we don't have the REPL in the console
print("\033[2J")

#endregion VEXcode Generated Robot Configuration

# Optical sensor setup
sensor.set_light(LedStateType.ON)

# Toggle state
motor_on = False

def toggle_motor():
    global motor_on
    motor_on = not motor_on
    if not motor_on:
        motor_1.stop()
        motor_10.stop()

# Set motor speed and torque to 100% for max volume
motor_1.set_velocity(100, PERCENT)
motor_10.set_velocity(100, PERCENT)
motor_1.set_max_torque(100, PERCENT)
motor_10.set_max_torque(100, PERCENT)

# 120BPM = 500ms per beat
tempo = 500  # msec

# Main loop
while True:
    brightness = sensor.brightness()  # Read sensor every loop (0-100)

    # Light present = play, darkness = stop
    if brightness > 10:
        motor_on = True
    else:
        motor_on = False

    if motor_on:
        # 4/4 Rhythm Loop: kick-snare-kick-snare

        # Beat 1: kick hit
        motor_10.spin_for(FORWARD, 90, DEGREES, wait=False)  # Strike immediately
        wait(tempo / 2, MSEC)                                # Wait half a beat
        motor_10.spin_for(REVERSE, 90, DEGREES, wait=True)   # Reset and wait for finish
        wait(tempo / 2, MSEC)                                # Wait the rest of the beat

        # Beat 2: snare hit
        motor_1.spin_for(FORWARD, 90, DEGREES, wait=False)
        wait(tempo / 2, MSEC)
        motor_1.spin_for(REVERSE, 90, DEGREES, wait=True)
        wait(tempo / 2, MSEC)

        # Beat 3: kick hit
        motor_10.spin_for(FORWARD, 90, DEGREES, wait=False)
        wait(tempo / 2, MSEC)
        motor_10.spin_for(REVERSE, 90, DEGREES, wait=True)
        wait(tempo / 2, MSEC)

        # Beat 4: snare hit
        motor_1.spin_for(FORWARD, 90, DEGREES, wait=False)
        wait(tempo / 2, MSEC)
        motor_1.spin_for(REVERSE, 90, DEGREES, wait=True)
        wait(tempo / 2, MSEC)

    else:
        motor_1.stop()
        motor_10.stop()
        wait(50, MSEC)
