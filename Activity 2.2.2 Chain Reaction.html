About

A motor control program using a potentiometer and a bumper switch. The motor only runs when the potentiometer tilt exceeds 45% — and stops permanently when the bumper is pressed, acting as an emergency shutoff. A live tilt percentage is displayed on the brain screen.

#region VEXcode Generated Robot Configuration
from vex import *
import urandom
import math

# Brain should be defined by default
brain=Brain()

# Robot configuration code
motor_5 = Motor(Ports.PORT5, GearSetting.RATIO_18_1, False)
bumper_b = Bumper(brain.three_wire_port.b)
potentiometer_a = Potentiometer(brain.three_wire_port.a)

# wait for rotation sensor to fully initialize
wait(30, MSEC)

# Make random actually random
def initializeRandomSeed():
    wait(100, MSEC)
    random = brain.battery.voltage(MV) + brain.battery.current(CurrentUnits.AMP) * 100 + brain.timer.system_high_res()
    urandom.seed(int(random))

# Set random seed 
initializeRandomSeed()

def play_vexcode_sound(sound_name):
    print("VEXPlaySound:" + sound_name)
    wait(5, MSEC)

wait(200, MSEC)
print("\033[2J")

is_enabled = True

# function to stop the motor indefinitely
def motor_stop():
    global is_enabled
    is_enabled = False  # stop
    motor_5.stop()

# bumper code outside loop so it doesnt tweak
bumper_b.pressed(motor_stop)

# the while true loop
while True:
    # live feed of potentiometer percentage
    brain.screen.set_cursor(1, 1)
    brain.screen.set_font(FontType.MONO20)
    brain.screen.set_pen_color(Color.YELLOW)
    brain.screen.print("Percent Tilt: ", potentiometer_a.angle(PERCENT), " %   ")

    # only run the motor if the potentiometer is > 45 AND the bumper hasn't been hit
    if is_enabled and potentiometer_a.angle(PERCENT) > 45:
        motor_5.spin(FORWARD, 25, PERCENT)
    else:
        # stops the motor if the potentiometer is low OR if is_enabled is False
        motor_5.stop()
